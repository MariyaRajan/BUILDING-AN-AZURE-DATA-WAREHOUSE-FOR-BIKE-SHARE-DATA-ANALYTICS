IF OBJECT_ID('dbo.staging_station') IS NOT NULL
    DROP EXTERNAL TABLE dbo.staging_station;
GO

CREATE EXTERNAL TABLE dbo.staging_station (
    station_id   varchar(50),
    station_name varchar(4000),
    latitude     varchar(50),
    longitude    varchar(50)
)
WITH (
    LOCATION = 'Source/public.station.csv',
    DATA_SOURCE = [divvyprojectcontainer1_divvyprojectstorageacc1_dfs_core_windows_net],
    FILE_FORMAT = [SynapseDelimitedTextFormat]
);
GO


SELECT TOP 10 * FROM dbo.staging_station;
GO
